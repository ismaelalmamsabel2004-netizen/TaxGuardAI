"use client"

import { useState, useCallback } from "react"
import { Shield, Upload, FileText, X } from "lucide-react"
import { Button } from "@/components/ui/button"
import { cn } from "@/lib/utils"

export function UploadSection() {
  const [isDragging, setIsDragging] = useState(false)
  const [uploadedFile, setUploadedFile] = useState<File | null>(null)

const enviarArchivo = async (archivo: File) => {
    const paquete = new FormData();
    paquete.append("file", archivo);

    try {
      const respuesta = await fetch("https://hook.eu1.make.com/dyfejas3ty576sa3b08kshf21w587dht", {
        method: "POST",
        body: paquete,
      });

      if (respuesta.ok) {
        alert("¡Documento recibido! La IA de TaxGuard está extrayendo los datos.");
      } else {
        alert("Fallo en la comunicación con el servidor.");
      }
    } catch (error) {
      console.error("Error en el túnel:", error);
    }
  };

  const handleDragOver = useCallback((e: React.DragEvent) => {
    e.preventDefault()
    setIsDragging(true)
  }, [])

  const handleDragLeave = useCallback((e: React.DragEvent) => {
    e.preventDefault()
    setIsDragging(false)
  }, [])

  const handleDrop = useCallback((e: React.DragEvent) => {
    e.preventDefault()
    setIsDragging(false)
    const files = e.dataTransfer.files
    if (files.length > 0) {
      setUploadedFile(files[0])
enviarArchivo(files[0]);
    }
  }, [])

  const handleFileChange = useCallback((e: React.ChangeEvent<HTMLInputElement>) => {
    const files = e.target.files
    if (files && files.length > 0) {
      setUploadedFile(files[0])
enviarArchivo(files[0]);
    }
  }, [])

  const clearFile = useCallback(() => {
    setUploadedFile(null)
  }, [])

  return (
    <section className="w-full">
      <div
        onDragOver={handleDragOver}
        onDragLeave={handleDragLeave}
        onDrop={handleDrop}
        className={cn(
          "relative overflow-hidden rounded-2xl border-2 border-dashed transition-all duration-300",
          isDragging
            ? "border-gold bg-gold/5 scale-[1.02]"
            : "border-gold/30 bg-card hover:border-gold/50 hover:bg-gold/5"
        )}
      >
        {/* Gradient overlay */}
        <div className="absolute inset-0 bg-gradient-to-br from-gold/5 via-transparent to-gold-light/5 pointer-events-none" />
        
        <div className="relative p-8 md:p-12">
          <div className="flex flex-col items-center text-center">
            {/* Shield Icon */}
            <div className="mb-6 flex h-20 w-20 items-center justify-center rounded-2xl bg-gold/10 border border-gold/30">
              <Shield className="h-10 w-10 text-gold" />
            </div>
            
            {uploadedFile ? (
              <div className="flex flex-col items-center gap-4">
                <div className="flex items-center gap-3 rounded-xl bg-muted px-4 py-3 border border-border">
                  <FileText className="h-5 w-5 text-gold" />
                  <span className="text-sm font-medium text-foreground">{uploadedFile.name}</span>
                  <button
                    onClick={clearFile}
                    className="ml-2 rounded-full p-1 hover:bg-background transition-colors"
                  >
                    <X className="h-4 w-4 text-muted-foreground" />
                  </button>
                </div>
                <Button 
                  size="lg"
                  className="bg-gold hover:bg-gold/90 text-background font-semibold px-8 shadow-lg shadow-gold/20"
                >
                  <Upload className="mr-2 h-4 w-4" />
                  Scan Now
                </Button>
              </div>
            ) : (
              <>
                <h2 className="mb-2 text-xl md:text-2xl font-semibold text-foreground">
                  Securely analyze any invoice or ticket
                </h2>
                <p className="mb-6 text-muted-foreground max-w-md">
                  Drag and drop your document here, or click to browse your files
                </p>
                
                <label htmlFor="file-upload" className="cursor-pointer">
                  <input
                    id="file-upload"
                    type="file"
                    className="sr-only"
                    accept=".pdf,.jpg,.jpeg,.png"
                    onChange={handleFileChange}
                  />
                  <Button 
                    asChild
                    size="lg"
                    className="bg-gold hover:bg-gold/90 text-background font-semibold px-8 shadow-lg shadow-gold/20"
                  >
                    <span>
                      <Upload className="mr-2 h-4 w-4" />
                      Upload Document
                    </span>
                  </Button>
                </label>
                
                <p className="mt-4 text-xs text-muted-foreground">
                  Supports PDF, JPG, PNG • Max 10MB
                </p>
              </>
            )}
          </div>
        </div>
      </div>
    </section>
  )
}
